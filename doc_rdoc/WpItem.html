def found_from=(method)
  found       = method[%{find_from_(.*)}, 1]
  @found_from = found.gsub('_', ' ') if found
end

def path=(path)
  @path = URI.encode(
    path.gsub(/\$wp-plugins\$/, wp_plugins_dir).gsub(/\$wp-content\$/, wp_content_dir)
  )
end
