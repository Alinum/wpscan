def vulnerabilities
  vulnerabilities = []

  xml = Nokogiri::XML(File.open(@vulns_file)) do |config|
    config.noblanks
  end

  xml.xpath(@vulns_xpath).each do |node|
    vulnerabilities << WpVulnerability.new(
      node.search("title").text,
      node.search("reference").map(&:text),
      node.search("type").text
    )
  end
  vulnerabilities
end
